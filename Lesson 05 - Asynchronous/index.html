Testing Asynchronous Code

In this section I learnt about working with and tesing promises. I also go to see how we can use the async functions and await keyword in unit tests. I set up a promise in the Calculator class to fetch the version from an external source which will be used in the below specs. 

  1. get version() {
  2.   return fetch('https://gist.githubusercontent.com/juanlizarazo/4b2d229ba483ca13b1a6d7bf3079dc8b/raw/228ac05e04118037be02c38d9b86945c1356a2e2/version.json').then(response => {
  3. 	return response.json()
  4.   }).then(data => {
  5. 	return data.version
  6.   })
  7. }

Since our getter version is now a promise we need to update the spec in the showVersion suite. 

  1. describe('showVersion()', () => {
  2.   it('calls calculator.version getter', () => {
  3. 	/**
  4. 	 * Here we can use install a spy on the DOM instead of using a setup and teardown approach
  5. 	*/
  6. 	spyOn(document, 'getElementById').and.returnValue({
  7. 	  id: 'calculator-version',
  8. 	  innerText: null,
  9. 	})
  10.  
  11. 	/**
  12. 	 * This installs a spy on the getter 'version' since spyOn cannot work with getters.
  13. 	*/
  14. 	const spy_on_getter = spyOnProperty(Calculator.prototype, 'version', 'get').and.returnValue(
  15. 	  /**
  16. 	   * Since we are now getting the version as a promise,
  17. 	   * we need to return a value as a resolved promise for the
  18. 	   * spy installed on the getter to work
  19. 	  */
  20. 	  Promise.resolve()
  21. 	)
  22.  
  23. 	showVersion()
  24.  
  25. 	expect(spy_on_getter).toHaveBeenCalled()
  26. 	expect(spy_on_getter).toHaveBeenCalledTimes(1)
  27.   })
  28. })
The 'done' callback

This callback is always passed to all specs by jasmine and is used to tell jasmine that there is an asynchronous function being tested and it should wait until the promise is resolved. Once the done callback is passed to the spec it is expected to be called otherwise jasmine will continue to run the spec and timeout. 

  1. describe('[get] version', () => {
  2.   it('fetches version from external source', (done) => {
  3. 	/**
  4. 	 * We can also install a spy on the fetch method used for the asynchronous request
  5. 	 * Here, we return a value that is a resolved promise with a response that contains the
  6. 	 * version value.
  7. 	*/
  8. 	spyOn(window, 'fetch').and.returnValue(Promise.resolve(
  9. 	  new Response('{"version": "0.1"}')
  10. 	))
  11.  
  12. 	/**
  13. 	 * Here, we place our expectation inside the promise since 
  14. 	 * we cannot have access to the version outside the promise
  15. 	 * 
  16. 	 * We use the done callback so as to let jasmine know that it
  17. 	 * should wait until the promise is resolved. This done callback is
  18. 	 * always passed by jasmine to all specs
  19. 	 * 
  20. 	 * Note: always call the done callback whenever it is passed to the spac as an argument
  21. 	*/
  22. 	calculator.version.then(version => {
  23. 	  expect(version).toBe('0.1')
  24.  
  25. 	  done()
  26. 	});
  27.   });
  28. });
Asyn and Await

Here, we convert the spec callback to an async function which allows us to use the await keyword on our promise. We still need the done callback to tell jasmine to wait for the promise to resolve. 

  1. describe('[get] version', () => {
  2.   /**
  3.    * The async callback lets jasmine know this is an asynchronous spec and this allows us to use the
  4.    * 'await' keyword which will wait for the version Promise to resolve
  5.   */
  6.   it('fetches version from external source', async (done) => {
  7. 	/**
  8. 	 * Here, we return a value that is a resolved promise with a response that contains the
  9. 	 * version value
  10. 	*/
  11. 	spyOn(window, 'fetch').and.returnValue(Promise.resolve(
  12. 	  new Response('{"version": "0.1"}')
  13. 	));
  14.  
  15. 	/**
  16. 	 * Here, because our suite is now an async function we can use the 'await' keyword
  17. 	 * and ommit the 'then' callback
  18. 	 * 
  19. 	 * Note: always call the done callback whenever it is passed to the spac as an argument
  20. 	*/
  21. 	const version = await calculator.version;
  22.  
  23. 	expect(version).toBe('0.1');
  24.  
  25. 	done();
  26.  
  27.   });
  28. });

Try it below:

Result: 3 Calculator Version 
  1. function calculate(input_value) {
  2.     /**
  3.      * Below regural expression(regex) is for matching our operators (+, -, * and /).
  4.      * In JavaScript we wrap regexes with a slash in the beginning and end of the expression.
  5.      * This regex was generated here: https://regex101.com/
  6.     */
  7.     const expression = /\+|\-|\*|\//;
  8.  
  9.     // Here we split the string from the input into an array
  10.     const numbers = input_value.split(expression);
  11.  
  12.     /**
  13.      * Get the first and second number in the arithmetic operation.
  14.      * Because they are strings we use parseInt() to convert them to numbers.
  15.     */
  16.     const first = parseInt(numbers[0]);
  17.     const second = parseInt(numbers[1]);
  18.  
  19.     /**
  20.      * The match helper is a string prototype that retrieves the result of matching a string against a
  21.      * regex. The result of it is returned as an array. In this case it will match the operator and return it.
  22.      * 
  23.      * More info: https://developer.mozilla.org/en-US/docs/Web/JavaScript/Reference/Global_Objects/String/match
  24.     */
  25.     const operator_match = input_value.match(expression);
  26.     let operator;
  27.  
  28.     if(Number.isNaN(first) || Number.isNaN(second) || operator_match === null) {
  29.         return updateResult('Operation not recognized');
  30.     } else {
  31.         // get the operator from the match result array
  32.         operator = operator_match[0];
  33.     }
  34.  
  35.     const calculator = new Calculator();
  36.     calculator.add(first);
  37.  
  38.     let result;
  39.  
  40.     switch (operator) {
  41.         case '+':
  42.             result = calculator.add(second);
  43.             break;
  44.         case '-':
  45.             result = calculator.subtract(second);
  46.             break;
  47.         case '*':
  48.             result = calculator.multiply(second);
  49.             break;
  50.         case '/':
  51.             result = calculator.divide(second);
  52.             break;
  53.         default:
  54.             break;
  55.     }
  56.  
  57.     return updateResult(result);
  58. };
  59.  
  60. function updateResult(calculator_result) {
  61.     const calculator_result_element = document.getElementById('calculator-result-span');
  62.  
  63.     if (calculator_result_element) {
  64.         calculator_result_element.innerText = `Result: ${calculator_result}`;
  65.     }
  66. };
  67.  
  68. function showVersion() {
  69.     const calculator =  new Calculator()
  70.  
  71.     const element = document.getElementById('calculator-version')
  72.  
  73.     // version is a getter in the Calculator class
  74.     calculator.version.then(version => {
  75.         element.innerText = version
  76.     })
  77. }
  78.  
  79. /**
  80.  * How to check an element exists in the DOM:
  81.  * 
  82.  * https://stackoverflow.com/questions/5629684/how-can-i-check-if-an-element-exists-in-the-visible-dom#answer-5629761
  83. */
  84. if (typeof(document.getElementById('calculator-input')) !== 'undefined' && document.getElementById('calculator-input') !== null) {
  85.     document.getElementById('calculator-input').addEventListener('keyup', (event) => {
  86.         if (event.key === 'Enter') {
  87.             return calculate(event.target.value);
  88.         }
  89.     })
  90. }
